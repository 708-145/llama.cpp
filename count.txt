Compile with 
cmake -B build -DBUILD_SHARED_LIBS=OFF
cmake --build build --config Release -j8 --target llama-cli

Test with Tiny-Moe.Q5_K.gguf
wget https://huggingface.co/mradermacher/Tiny-Moe-GGUF/resolve/main/Tiny-Moe.Q5_K.gguf

./llama-cli -p "What is 3+4?" -n 32 -m Tiny-Moe.Q5_K.gguf
